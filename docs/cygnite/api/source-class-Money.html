 42:         $this->currency = $curr;

 45:     function add( $money ) {
 46:     // the inherited, 'overridden' method....
 47:     // using Double Dispatch return the passed money object
 48:     // after having called it's addMoney method passing 'this' == self as a parameter.
 49: 
 50:         return $money->addMoney( $this );
 51:     }

 53:     function addMoney( $money ) {
 54:     // the inherited, 'overridden' method....
 55:     // return a new object of appropriate type 
 56:     // based on the parameter type passed in..
 57: 
 58:         if ( $this->currency() == $money->currency() ) {
 59:             // add if money is same currency, then return a new money object.
 60:             $result = new Money($this->amount() + $money->amount(), $this->currency() );
 61:         } else { 
 62:             // other wise return a new money bag passing 'this' ==self as a parameter...
 63:             $result = new MoneyBag( $this, $money); 
 64:         } 
 65:         return $result;
 66:     }
 67: 
 68:     function addMoneyBag( $moneyBag ) {
 69:     // the inherited, 'overridden' method....

 72:     }

115:         // the first parameter can be either an array of monies, a money object, or a money bag object...

118:         // deal with the first parameter...

125:         }

127:         // deal with the second parameter...

138:     // a method that will load the moneybag objects allMyMonies[ ] collection/array
139:     // of money objects
140: 
141:         $size = sizeof( $moneyArray );
142:         for ($count=0; $count<$size; $count++) {
143:             $this->appendMoney( $moneyArray[ $count ] );
144:         }
145:     }

155:         }

159:     // the inherited, 'overridden' method....

161:     // after having called it's addMoneyBag method passing 'this' == self as a parameter.

167:     // the inherited, 'overridden' method....

174:         return new MoneyBag( $moneyBag, $this );

177:     function appendMoney( $money ) {
178:         // add this money to the money bags monies...
179:         // if the currency already exists, then increase the total of this currency..
180: 
181:         // a flag...
182:         $added = FALSE; 
183:         $size = $this->size();
184: 
185:         for ( $count=0; $count<$size; $count++) {
186:             if ( $added == FALSE ) {
187:                 if ( $this->allMyMonies[$count]->currency() == $money->currency() ) {
188:                     $this->allMyMonies[$count] = $money->add( $this->allMyMonies[$count] );
189:                     $added = TRUE;
190:                 } 
191:             }
192:         }
193: 
194:         // after having check all existing monies, and still not added,
195:         // add the money to the bag...

199:         }

201:     }
