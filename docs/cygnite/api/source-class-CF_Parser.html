 61:       public function safefilerewrite($file_name, $data_to_save)
 62:      {
 63:           if ($fp = fopen($file_name, 'w')):
 64:             $startTime = microtime();
 65:             do
 66:             {
 67:                 $canWrite = flock($fp, LOCK_EX);
 68:                // If lock not obtained sleep for 0 - 100 milliseconds, to avoid collision and CPU load
 69:                if(!$canWrite) usleep(round(rand(0, 100)*1000));
 70:             } while ((!$canWrite)and((microtime()-$startTime) < 1000));
 71: 
 72:                 //file was locked so now we can store information
 73:                 if ($canWrite):
 74:                     fwrite($fp, $data_to_save);
 75:                     flock($fp, LOCK_UN);
 76:                 endif;
 77:               fclose($fp);
 78:         endif;
 79:       }
